40static int cstr_ccmp(const char * str1, const char * str2){
41 char * s1 = (char *) str1;
42 char * s2 = (char *) str2;
43 while (*s1 != '\0' && *s2 != '\0' && cto_lower(*s1) == cto_lower(*s2)){
44 s2++;
45 s1++;
46 }
47 return cto_lower(*s1) - cto_lower(*s2);
48}

76static unsigned int scstr_count(const char * str, int ch){
77 // counts the number of occurrence of ch in str
78 // str must not be null

80 char * tmp = (char*)str;
81 while (*tmp){
82 if (*tmp == ch)

118static ctld_result * ctld_choose_best(PCLIST all_match, char * domain){
119 ctld_node * best = NULL;
120 ctld_node * item = NULL;
121 unsigned long int level = 0;
122 unsigned long int tmp_level = 0;
123 unsigned long int domain_len = strlen(domain);
124 unsigned long int tmp_len = 0;
125 for (int i=0; i<= all_match->last_item; ++i){
126 item = (ctld_node *)all_match->list[i];
127 tmp_len = strlen(item->name);
128 tmp_level = scstr_count(item->name, '.') + 1;
129 if (item->has_priority){ // get the one with the priority
130 best = item;
131 break;
132 }
133 if ( tmp_level> level || best == NULL){
134 if (item->name[0] == '*') // we don't match as * will be empty
135 if (domain_len <= tmp_len - 2)
136 continue;
137 level = tmp_level;
138 best = item;
139 continue;
140 }
141 }
142 if (!best || !best->name)
143 return NULL;
144 // we have one candidate
145 char * tmp = NULL;
146 tmp = best->name[0] == '*'?strdup(best->name + 2):strdup(best->name);
147 ctld_result * result = (ctld_result*) malloc(sizeof(ctld_result));
148 if (!result){
149 free(tmp);
150 return NULL;
151 }
152 result->suffix = NULL;
153 result->registered_domain = NULL;
154 result->fqdn = NULL;
155 result->domain = NULL;
156 PSTR str = str_init(tmp);
157 free(tmp);
158 PSTR str_domain = str_init(domain);
159 PSPLITLIST str_split = NULL;
160 int max_split;
161 // make sure to split based on dot but not the suffix part
162 max_split = str_domain->str_count(str_domain, ".") - str->str_count(str, ".");
163 str_split = str_domain->str_split(str_domain, ".", max_split);
164 if (best->has_priority){
165 // first part is domain name and the rest is tld
166 PSPLITLIST str_tmp_split = str->str_split(str, ".", 1);
167 result->suffix = strdup(str_tmp_split->list[1]);
168 str->str_setval(str, result->suffix);
169 str_free_splitlist(str_tmp_split);
170 }else if (best->name[0] == '*'){
171 str->str_prepend_char(str, '.');
172 str->str_prepend_string(str, str_split->list[str_split->len -2]);
173 result->suffix = str->str_copy(str);
174 }else{
175 // suffix is exactly what it is
176 result->suffix = str->str_copy(str);
177 }
178 // now the suffix might be different and we need to re-split
179 str_free_splitlist(str_split);
180 max_split = str_domain->str_count(str_domain, ".") - str->str_count(str, ".");
181 str_split = str_domain->str_split(str_domain, ".", max_split);
182
183 result->domain = str_split->len <= 1?NULL:strdup(str_split->list[str_split->len -2]);
184 unsigned long int len_domain = 0, len_suffix = 0;
185 len_suffix = str->len;
186 str_free_splitlist(str_split);
187
188 str_free(str);
189 if (result->domain){
190 len_domain = strlen(result->domain);
191 result->registered_domain = (char*) malloc(len_domain + len_suffix + 2);
192 }
193 if (result->registered_domain){
194 memcpy(result->registered_domain, result->domain, len_domain);
195 memcpy(result->registered_domain + len_domain, (const char*)".", 1);
196 memcpy(result->registered_domain + len_domain + 1, result->suffix, len_suffix);
197 result->registered_domain[len_domain + len_suffix + 1] = '\0';
198 }
199 str_free(str_domain);
200 result->fqdn = result->domain?strdup(domain):NULL;
201 return result;
202}

213 PSPLITLIST plst = str->str_split(str, "\n", -1);
214 if (!plst){
215#ifdef DEBUG
216 fprintf(stdout, "Can not split the list\n");
217#endif
218 return 1;

220 PSTR line = str_init(NULL);
221 str_free(str);
222 char * stripped_line = NULL;

240 end_private_part = 1;

242 continue;

244 continue;

306 if (idna_out && cstr_ccmp(idna_out, new_node->name) != 0){
307 // here we have IDNA, we need to add a new node for this

478int ctld_is_domain_valid(char * domain){
479 // simple checks to see if the fully-qualified domain is valid
480 // the domain must have atleast one dot
481 // at least 4 characters
482 // no special character
483 // returns 1 if valid 0 if invalid
484 if (!domain)
485 return 0;
486 PSTR str = str_init(domain);
487 if (str->str_len(str) < 4){
488 str_free(str);
489 return 0;
490 }
491 if (str->str_find(str, ".") == -1){
492 str_free(str);
493 return 0;
494 }
495 if (str->str_any_of_in(str, "!@#$%^&*()_+=\\/,|'\"")){
496 str_free(str);
497 return 0;
498 }
499 str_free(str);
500 return 1;
501}

516ctld_result * ctld_parse(ctld_ctx * ctx, char * domain, int use_private_suffix){
517 if (!domain || !ctx)
518 return NULL;
519 // do we really care if the fqdn is correct or not?
520 //if (ctld_is_domain_valid(domain) != 1)
521 // return NULL;
522 ctld_node * match_public, *match_private = NULL;
523 PSTR str = str_init(domain);
524 PSTR new_str = str_init(NULL);
525 // just search for a perfect match in list
526 PCLIST all_match = clist_init(ctld_node_free);
527
528 match_public = ctld_search_in_list(ctx->list_public, domain);
529 match_private = use_private_suffix?ctld_search_in_list(ctx->list_private, domain):NULL;
530 if (match_public)
531 clist_add_item(all_match, (void*)match_public);
532 if (match_private && use_private_suffix)
533 clist_add_item(all_match, (void*) match_private);
534 new_str->str_setval(new_str, "*.");
535 new_str->str_append_string(new_str, domain);
536 match_public = ctld_search_in_list(ctx->list_public, new_str->str);
537 match_private = use_private_suffix?ctld_search_in_list(ctx->list_private, new_str->str):NULL;
538 if (match_public)
539 clist_add_item(all_match, (void*) match_public);
540 if (match_private && use_private_suffix)
541 clist_add_item(all_match, (void*)match_private);
542
543 PSPLITLIST str_split = str->str_split(str, ".", -1);
544 SPLITLIST new_split;
545 for (int i=1; i<str_split->len && str_split->len > 1; ++i){
546 new_split.len = str_split->len -i;
547 new_split.list = str_split->list + i;
548 new_str->str_join(new_str, &new_split, ".");
549 match_public = ctld_search_in_list(ctx->list_public, new_str->str);
550 match_private = use_private_suffix?ctld_search_in_list(ctx->list_private, new_str->str):NULL;
551 if (match_public)
552 clist_add_item(all_match, (void*) match_public);
553 if (use_private_suffix && match_private)
554 clist_add_item(all_match, (void*)match_private);
555
556 new_str->str_prepend_string(new_str, "*.");
557 match_public = ctld_search_in_list(ctx->list_public, new_str->str);
558 match_private = use_private_suffix?ctld_search_in_list(ctx->list_private, new_str->str):NULL;
559 if (match_public)
560 clist_add_item(all_match, (void*) match_public);
561 if (use_private_suffix && match_private)
562 clist_add_item(all_match, (void*)match_private);
563
564 }
565
566 str_free_splitlist(str_split);
567 str_free(new_str);
568 str_free(str);
569 if (!all_match || all_match->first_empty == 0){
570 free(all_match->list);
571 free(all_match);
572 ctx->errcode = CTLD_NO_MATCH_FOUND;
573#ifdef DEBUG
574 fprintf(stdout, "Can not find a match for a given domain: %s\n", domain);
575#endif
576 return NULL;
577 }
578 ctld_result * result = ctld_choose_best(all_match, domain);
579 // we can not call clist_free() on all match as
580 // we have refernce not copy of the data
581 // we just need to free the list and the structure itself
582 // not the content inside the list
583 free(all_match->list);
584 free(all_match);
585 return result;
586}
